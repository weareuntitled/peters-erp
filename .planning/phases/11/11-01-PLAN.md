Create mutation:
```typescript
const queryClient = useQueryClient();
const createDocMutation = useMutation({
  mutationFn: (data: any) => apiClient.post('/dokumente', data),
  onSuccess: (response) => {
    queryClient.invalidateQueries({ queryKey: ['dokumente'] });
    queryClient.invalidateQueries({ queryKey: ['angebote'] });
    queryClient.invalidateQueries({ queryKey: ['rechnungen'] });
    const docType = response.data.typ;
    navigate(docType === 'AN' ? '/angebote' : '/rechnungen');
  },
});
```

Build payload in handleSave from current state:
```typescript
const handleSave = () => {
  if (!selectedKunde) return; // guard: must select customer
  const payload = {
    typ: docType,
    kunde_id: selectedKunde.id,
    datum: datum,
    gueltigkeit: parseInt(gueltigkeit, 10),
    projekt: projekt || null,
    positionen: positionen.map((p, idx) => ({
      position_nr: idx + 1,

      menge: p.menge,
      einheit: p.einheit,
      einzelpreis: p.einzelpreis,
    })),
  };
  createDocMutation.mutate(payload);
};
```
